package com.simplelearnprojects.phase1;
class Overloading{
	int add(int a, int b) {
		return(a+b);
	}
	int add(int a, int b, int c) {
		return(a+b-c);
	}
}

public class Methods {

	public static void main(String[] args) {
		Overloading C = new Overloading();
		System.out.println(C.add(1,2,3));

	}

}
